import React, { useState } from "react";
import { Navigate, Route, Routes } from 'react-router-dom'
import Welcome from "./components/Welcome";
import AddTask from "./components/AddTask";
import List from "./components/List";

export default function App() {
  const [tasks, setTasks] = useState([]);

  return (
    <div style={{ fontFamily: "Arial", margin: "20px" }}>
      <nav>
        <Link to="/">Welcome</Link> |{" "}
        <Link to="/add">Add Task</Link> |{" "}
        <Link to="/list">Task List</Link>
      </nav>
      <hr />
      <Routes>
        <Route path="/" element={<Welcome />} />
        <Route path="/add" element={<AddTask tasks={tasks} setTasks={setTasks} />} />
        <Route path="/list" element={<List tasks={tasks} />} />
      </Routes>
    </div>
  );
}


import React, { useState } from 'react'
import { useNavigate } from 'react-router-dom'

function Welcome() {
    const handleclick = () =>{
        navigate("/add");
    }
    const handleclicktasklist = () =>{
        navigate("/list");
    }
    const  navigate= useNavigate();

  return (
    <div>
        <h1>Welcome to  to-do items system </h1>
        <button type="button" onClick={handleclick}>Add Task</button>
        <button type="button" onClick={handleclicktasklist}>Task List</button>

    </div>
  )
}

export default Welcome





import React, { useState } from "react";
import { useNavigate } from "react-router-dom";

export default function AddTask({ tasks, setTasks }) {
  const [name, setName] = useState("");
  const [priority, setPriority] = useState("");
  const [date, setDate] = useState("");
  const [message, setMessage] = useState("");
  const navigate = useNavigate();

  // Get today's date in yyyy-mm-dd format
  const today = new Date().toISOString().slice(0, 10);

  const handleSubmit = (e) => {
    e.preventDefault();

    if (!name.trim() || !priority) {
      alert("Please fill in required fields!");
      return;
    }

    if (date && new Date(date) < new Date(today)) {
      alert("Due date cannot be in the past!");
      return;
    }

    const newTask = { name, priority, date };
    setTasks([...tasks, newTask]);
    setMessage(
      `Task '${name}' with priority ${priority} added. Keep up the good work!`
    );

    // Reset fields
    setName("");
    setPriority("");
    setDate("");

    // Redirect to Task List
    navigate("/list");
  };

  return (
    <div>
      <h2>Add Task</h2>
      <form onSubmit={handleSubmit}>
        <div>
          <label>
            Task Name: <br />
            <input
              type="text"
              value={name}
              onChange={(e) => setName(e.target.value)}
              required
            />
          </label>
        </div>
        <div>
          <label>
            Priority: <br />
            <select
              value={priority}
              onChange={(e) => setPriority(e.target.value)}
              required
            >
              <option value="">--Select Priority--</option>
              <option>Low</option>
              <option>Medium</option>
              <option>High</option>
            </select>
          </label>
        </div>
        <div>
          <label>
            Due Date: <br />
            <input
              type="date"
              value={date}
              onChange={(e) => setDate(e.target.value)}
              min={today} // Prevent past dates
            />
          </label>
        </div>
        <br />
        <button type="submit">Add Task</button>
      </form>
      {message && <p style={{ color: "green" }}>{message}</p>}
    </div>
  );
}




import React from "react";

export default function List({ tasks }) {
  if (tasks.length === 0) return <p>No tasks yet. Add some!</p>;

  return (
    <div>
      <h2>Task List</h2>
      {tasks.map((task, index) => (
        <div key={index}>
          {index + 1}. {task.name} | Priority: {task.priority} | Due: {task.date || "Not set"}
        </div>
      ))}
    </div>
  );
}

  </StrictMode>,
)
